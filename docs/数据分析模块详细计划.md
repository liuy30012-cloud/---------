<script setup lang="ts">
import { ref, onMounted, onUnmounted, nextTick } from 'vue'
import * as echarts from 'echarts'
import axios from 'axios'
import { API_CONFIG } from '../config'

// 当前时间
const currentTime = ref('')
let timeInterval: ReturnType<typeof setInterval>

// 关键指标
const keyMetrics = ref([
  { icon: '📚', label: '馆藏总量', value: '0', change: '+0', trend: 'up' },
  { icon: '📖', label: '今日借阅', value: '0', change: '+0', trend: 'up' },
  { icon: '↩️', label: '今日归还', value: '0', change: '+0', trend: 'up' },
  { icon: '👥', label: '活跃用户', value: '0', change: '+0', trend: 'up' }
])

// 热门书籍
const topBooks = ref<any[]>([])

// 实时动态
const recentActivities = ref<any[]>([])

// 图表引用
const trendChartRef = ref<HTMLElement>()
const categoryChartRef = ref<HTMLElement>()
const userActivityChartRef = ref<HTMLElement>()

let trendChart: echarts.ECharts
let categoryChart: echarts.ECharts
let userActivityChart: echarts.ECharts

// 更新时间
function updateTime() {
  const now = new Date()
  currentTime.value = now.toLocaleString('zh-CN', {
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit',
    hour12: false
  })
}

// 加载数据
async function loadDashboardData() {
  try {
    const response = await axios.get(`${API_CONFIG.baseURL}/api/statistics/dashboard`)
    const data = response.data

    // 更新关键指标
    keyMetrics.value[0].value = data.totalBooks?.toLocaleString() || '0'
    keyMetrics.value[1].value = data.todayBorrows?.toString() || '0'
    keyMetrics.value[2].value = data.todayReturns?.toString() || '0'
    keyMetrics.value[3].value = data.activeUsers?.toString() || '0'

    // 更新热门书籍
    topBooks.value = data.topBooks || []

    // 渲染图表
    renderTrendChart(data.recentTrends || [])
    renderCategoryChart(data.categoryStats || [])
    renderUserActivityChart(data.userActivity || [])

    // 模拟实时动态
    generateRecentActivities()
  } catch (error) {
    console.error('Failed to load dashboard data:', error)
  }
}

// 渲染借阅趋势图
function renderTrendChart(data: any[]) {
  if (!trendChart && trendChartRef.value) {
    trendChart = echarts.init(trendChartRef.value)
  }

  const option = {
    tooltip: {
      trigger: 'axis',
      backgroundColor: 'rgba(0, 0, 0, 0.8)',
      borderColor: '#667eea',
      textStyle: {
        color: '#fff'
      }
    },
    grid: {
      left: '3%',
      right: '4%',
      bottom: '3%',
      top: '10%',
      containLabel: true
    },
    xAxis: {
      type: 'category',
      data: data.map(item => item.date),
      axisLine: {
        lineStyle: {
          color: '#667eea'
        }
      },
      axisLabel: {
        color: '#fff'
      }
    },
    yAxis: {
      type: 'value',
      axisLine: {
        lineStyle: {
          color: '#667eea'
        }
      },
      axisLabel: {
        color: '#fff'
      },
      splitLine: {
        lineStyle: {
          color: 'rgba(102, 126, 234, 0.2)'
        }
      }
    },
    series: [
      {
        name: '借阅量',
        type: 'line',
        smooth: true,
        data: data.map(item => item.count),
        areaStyle: {
          color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
            { offset: 0, color: 'rgba(102, 126, 234, 0.8)' },
            { offset: 1, color: 'rgba(102, 126, 234, 0.1)' }
          ])
        },
        lineStyle: {
          color: '#667eea',
          width: 3
        },
        itemStyle: {
          color: '#667eea'
        }
      }
    ]
  }

  trendChart.setOption(option)
}

// 渲染分类占比图
function renderCategoryChart(data: any[]) {
  if (!categoryChart && categoryChartRef.value) {
    categoryChart = echarts.init(categoryChartRef.value)
  }

  const colors = ['#667eea', '#764ba2', '#f093fb', '#4facfe', '#43e97b']

  const option = {
    tooltip: {
      trigger: 'item',
      formatter: '{a} <br/>{b}: {c} ({d}%)',
      backgroundColor: 'rgba(0, 0, 0, 0.8)',
      borderColor: '#667eea',
      textStyle: {
        color: '#fff'
      }
    },
    legend: {
      orient: 'vertical',
      right: '10%',
      top: 'center',
      textStyle: {
        color: '#fff'
      }
    },
    series: [
      {
        name: '借阅量',
        type: 'pie',
        radius: ['40%', '70%'],
        center: ['35%', '50%'],
        avoidLabelOverlap: false,
        itemStyle: {
          borderRadius: 10,
          borderColor: '#1a1a2e',
          borderWidth: 2
        },
        label: {
          show: false
        },
        emphasis: {
          label: {
            show: true,
            fontSize: 16,
            fontWeight: 'bold',
            color: '#fff'
          }
        },
        labelLine: {
          show: false
        },
        data: data.map((item, index) => ({
          name: item.category,
          value: item.borrowCount,
          itemStyle: {
            color: colors[index % colors.length]
          }
        }))
      }
    ]
  }

  categoryChart.setOption(option)
}

// 渲染用户活跃度图
function renderUserActivityChart(data: any[]) {
  if (!userActivityChart && userActivityChartRef.value) {
    userActivityChart = echarts.init(userActivityChartRef.value)
  }

  const option = {
    tooltip: {
      trigger: 'item',
      backgroundColor: 'rgba(0, 0, 0, 0.8)',
      borderColor: '#667eea',
      textStyle: {
        color: '#fff'
      }
    },
    series: [
      {
        name: '用户类型',
        type: 'pie',
        radius: '70%',
        data: [
          { value: 335, name: '活跃用户', itemStyle: { color: '#43e97b' } },
          { value: 234, name: '普通用户', itemStyle: { color: '#667eea' } },
          { value: 135, name: '不活跃', itemStyle: { color: '#f093fb' } }
        ],
        label: {
          color: '#fff',
          fontSize: 12
        },
        emphasis: {
          itemStyle: {
            shadowBlur: 10,
            shadowOffsetX: 0,
            shadowColor: 'rgba(0, 0, 0, 0.5)'
          }
        }
      }
    ]
  }

  userActivityChart.setOption(option)
}

// 生成实时动态
function generateRecentActivities() {
  const activities = [
    { id: 1, type: 'borrow', icon: '📖', text: '用户 张三 借阅了《代码整洁之道》', time: '刚刚' },
    { id: 2, type: 'return', icon: '↩️', text: '用户 李四 归还了《算法导论》', time: '2分钟前' },
    { id: 3, type: 'borrow', icon: '📖', text: '用户 王五 借阅了《设计模式》', time: '5分钟前' },
    { id: 4, type: 'return', icon: '↩️', text: '用户 赵六 归还了《深入理解计算机系统》', time: '8分钟前' },
    { id: 5, type: 'borrow', icon: '📖', text: '用户 钱七 借阅了《百年孤独》', time: '10分钟前' }
  ]
  recentActivities.value = activities
}

// 自动刷新数据
let refreshInterval: ReturnType<typeof setInterval>

onMounted(async () => {
  await nextTick()
  
  // 更新时间
  updateTime()
  timeInterval = setInterval(updateTime, 1000)

  // 加载数据
  await loadDashboardData()

  // 自动刷新（每30秒）
  refreshInterval = setInterval(loadDashboardData, 30000)

  // 响应式调整
  window.addEventListener('resize', () => {
    trendChart?.resize()
    categoryChart?.resize()
    userActivityChart?.resize()
  })
})

onUnmounted(() => {
  clearInterval(timeInterval)
  clearInterval(refreshInterval)
  trendChart?.dispose()
  categoryChart?.dispose()
  userActivityChart?.dispose()
})
</script>
