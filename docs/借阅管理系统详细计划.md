## 🎯 功能需求

### 核心功能
1. **借阅功能** - 在线借阅申请、审核、期限管理、续借
2. **归还功能** - 归还登记、逾期计算、罚款管理
3. **预约功能** - 书籍预约、预约队列、到馆通知

### 业务规则
- 每个用户最多同时借阅 **5本书**
- 默认借阅期限 **30天**
- 可续借 **1次**，续借延长 **15天**
- 逾期罚款 **0.5元/天**
- 预约保留时间 **3天**

### 1. 借阅记录表 (borrow_records)

```sql
CREATE TABLE borrow_records (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL COMMENT '用户ID',
    book_id BIGINT NOT NULL COMMENT '书籍ID',
    student_id VARCHAR(20) NOT NULL COMMENT '学号',
    book_title VARCHAR(200) NOT NULL COMMENT '书名',
    book_isbn VARCHAR(50) COMMENT 'ISBN',
    
    -- 借阅信息
    borrow_date DATETIME NOT NULL COMMENT '借阅日期',
    due_date DATETIME NOT NULL COMMENT '应还日期',
    return_date DATETIME COMMENT '实际归还日期',
    
    -- 状态管理
    status VARCHAR(20) NOT NULL COMMENT '状态: PENDING(待审核), APPROVED(已批准), BORROWED(借阅中), RETURNED(已归还), OVERDUE(已逾期), REJECTED(已拒绝)',
    
    -- 续借信息
    renewed BOOLEAN DEFAULT FALSE COMMENT '是否已续借',
    renew_count INT DEFAULT 0 COMMENT '续借次数',
    
    -- 逾期信息
    overdue_days INT DEFAULT 0 COMMENT '逾期天数',
    fine_amount DECIMAL(10,2) DEFAULT 0.00 COMMENT '罚款金额',
    fine_paid BOOLEAN DEFAULT FALSE COMMENT '罚款是否已支付',
    
    -- 审核信息
    approved_by VARCHAR(50) COMMENT '审核人',
    approved_at DATETIME COMMENT '审核时间',
    reject_reason VARCHAR(500) COMMENT '拒绝原因',
    
    -- 备注
    notes TEXT COMMENT '备注',
    
    -- 时间戳
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    INDEX idx_user_id (user_id),
    INDEX idx_book_id (book_id),
    INDEX idx_student_id (student_id),
    INDEX idx_status (status),
    INDEX idx_due_date (due_date)
) COMMENT='借阅记录表';
```

```java
    private Boolean finePaid = false;

    @Column(name = "approved_by", length = 50)
    private String approvedBy;
    
    @Column(name = "approved_at")
    private LocalDateTime approvedAt;

    private String rejectReason;
    
    // 备注
    @Column(columnDefinition = "TEXT")
    private String notes;

    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;
```

### 4. 服务层 (Service)

#### BorrowService.java

```java
package com.library.service;

import com.library.dto.BorrowRequest;
import com.library.dto.BorrowResponse;
import com.library.model.Book;
import com.library.model.BorrowRecord;
import com.library.model.BorrowRecord.BorrowStatus;
import com.library.model.User;
import com.library.repository.BorrowRecordRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.time.temporal.ChronoUnit;
import java.util.List;
import java.util.stream.Collectors;

@Slf4j
@Service
@RequiredArgsConstructor
public class BorrowService {
    
    private final BorrowRecordRepository borrowRecordRepository;
    private final BookService bookService;
    private final UserService userService;
    private final NotificationService notificationService;
    
    // 业务常量
    private static final int MAX_BORROW_COUNT = 5;
    private static final int DEFAULT_BORROW_DAYS = 30;
    private static final int RENEW_DAYS = 15;
    private static final int MAX_RENEW_COUNT = 1;
    private static final BigDecimal FINE_PER_DAY = new BigDecimal("0.50");
    
    /**
     * 申请借阅
     */
    @Transactional
    public BorrowResponse applyBorrow(Long userId, BorrowRequest request) {
        // 1. 验证用户
        User user = userService.getUserById(userId);
        if (user == null) {
            throw new IllegalArgumentException("用户不存在");
        }
        
        // 2. 验证书籍
        Book book = bookService.getBookById(request.getBookId());
        if (book == null) {
            throw new IllegalArgumentException("书籍不存在");
        }
        
        // 3. 检查借阅数量限制
        long currentBorrowCount = borrowRecordRepository.countCurrentBorrowsByUserId(userId);
        if (currentBorrowCount >= MAX_BORROW_COUNT) {
            throw new IllegalArgumentException("您已达到最大借阅数量限制（" + MAX_BORROW_COUNT + "本）");
        }
        
        // 4. 检查书籍库存
        if (book.getAvailableCopies() <= 0) {
            throw new IllegalArgumentException("该书籍暂无可借副本");
        }
        
        // 5. 创建借阅记录
        BorrowRecord record = new BorrowRecord();
        record.setUserId(userId);
        record.setBookId(book.getId());
        record.setStudentId(user.getStudentId());
        record.setBookTitle(book.getTitle());
        record.setBookIsbn(book.getIsbn());
        record.setBorrowDate(LocalDateTime.now());
        record.setDueDate(LocalDateTime.now().plusDays(DEFAULT_BORROW_DAYS));
        record.setStatus(BorrowStatus.PENDING);
        record.setNotes(request.getNotes());
        
        BorrowRecord saved = borrowRecordRepository.save(record);
        
        // 6. 发送通知
        notificationService.sendBorrowApplicationNotification(user, book);
        
        log.info("用户 {} 申请借阅书籍 {}", user.getStudentId(), book.getTitle());
        
        return convertToResponse(saved);
    }
    
    /**
     * 审核借阅申请
     */
    @Transactional
    public BorrowResponse approveBorrow(Long recordId, String approver, boolean approved, String rejectReason) {
        BorrowRecord record = borrowRecordRepository.findById(recordId)
            .orElseThrow(() -> new IllegalArgumentException("借阅记录不存在"));
        
        if (record.getStatus() != BorrowStatus.PENDING) {
            throw new IllegalArgumentException("该借阅申请已处理");
        }
        
        if (approved) {
            // 批准借阅
            Book book = bookService.getBookById(record.getBookId());
            if (book.getAvailableCopies() <= 0) {
                throw new IllegalArgumentException("书籍已无可借副本");
            }
            
            record.setStatus(BorrowStatus.BORROWED);
            record.setApprovedBy(approver);
            record.setApprovedAt(LocalDateTime.now());
            
            // 减少可借数量
            bookService.decreaseAvailableCopies(book.getId());
            
            log.info("借阅申请 {} 已批准", recordId);
        } else {
            // 拒绝借阅
            record.setStatus(BorrowStatus.REJECTED);
            record.setRejectReason(rejectReason);
            
            log.info("借阅申请 {} 已拒绝: {}", recordId, rejectReason);
        }
        
        BorrowRecord saved = borrowRecordRepository.save(record);
        
        // 发送通知
        User user = userService.getUserById(record.getUserId());
        notificationService.sendBorrowApprovalNotification(user, record);
        
        return convertToResponse(saved);
    }
    
    /**
     * 归还书籍
     */
    @Transactional
    public BorrowResponse returnBook(Long recordId) {
        BorrowRecord record = borrowRecordRepository.findById(recordId)
            .orElseThrow(() -> new IllegalArgumentException("借阅记录不存在"));
        
        if (record.getStatus() != BorrowStatus.BORROWED && record.getStatus() != BorrowStatus.OVERDUE) {
            throw new IllegalArgumentException("该书籍不在借阅中");
        }
        
        // 1. 计算逾期
        LocalDateTime now = LocalDateTime.now();
        if (now.isAfter(record.getDueDate())) {
            long overdueDays = ChronoUnit.DAYS.between(record.getDueDate(), now);
            record.setOverdueDays((int) overdueDays);
            record.setFineAmount(FINE_PER_DAY.multiply(new BigDecimal(overdueDays)));
        }
        
        // 2. 更新记录
        record.setReturnDate(now);
        record.setStatus(BorrowStatus.RETURNED);
        
        BorrowRecord saved = borrowRecordRepository.save(record);
        
        // 3. 增加可借数量
        bookService.increaseAvailableCopies(record.getBookId());
        
        // 4. 发送通知
        User user = userService.getUserById(record.getUserId());
        notificationService.sendReturnNotification(user, record);
        
        log.info("用户 {} 归还书籍 {}", record.getStudentId(), record.getBookTitle());
        
        return convertToResponse(saved);
    }
    
    /**
     * 续借
     */
    @Transactional
    public BorrowResponse renewBorrow(Long recordId) {
        BorrowRecord record = borrowRecordRepository.findById(recordId)
            .orElseThrow(() -> new IllegalArgumentException("借阅记录不存在"));
        
        // 1. 验证状态
        if (record.getStatus() != BorrowStatus.BORROWED) {
            throw new IllegalArgumentException("只能续借借阅中的书籍");
        }
        
        // 2. 验证续借次数
        if (record.getRenewCount() >= MAX_RENEW_COUNT) {
            throw new IllegalArgumentException("已达到最大续借次数");
        }
        
        // 3. 验证是否逾期
        if (LocalDateTime.now().isAfter(record.getDueDate())) {
            throw new IllegalArgumentException("书籍已逾期，无法续借");
        }
        
        // 4. 检查是否有预约
        // TODO: 如果有人预约该书，不允许续借
        
        // 5. 续借
        record.setDueDate(record.getDueDate().plusDays(RENEW_DAYS));
        record.setRenewed(true);
        record.setRenewCount(record.getRenewCount() + 1);
        
        BorrowRecord saved = borrowRecordRepository.save(record);
        
        // 6. 发送通知
        User user = userService.getUserById(record.getUserId());
        notificationService.sendRenewNotification(user, record);
        
        log.info("用户 {} 续借书籍 {}", record.getStudentId(), record.getBookTitle());
        
        return convertToResponse(saved);
    }
    
    /**
     * 查询用户借阅历史
     */
    public List<BorrowResponse> getUserBorrowHistory(Long userId) {
        List<BorrowRecord> records = borrowRecordRepository.findByUserIdOrderByCreatedAtDesc(userId);
        return records.stream()
            .map(this::convertToResponse)
            .collect(Collectors.toList());
    }
    
    /**
     * 查询用户当前借阅
     */
    public List<BorrowResponse> getUserCurrentBorrows(Long userId) {
        List<BorrowRecord> records = borrowRecordRepository.findByUserIdAndStatus(userId, BorrowStatus.BORROWED);
        return records.stream()
            .map(this::convertToResponse)
            .collect(Collectors.toList());
    }
    
    /**
     * 检查逾期（定时任务调用）
     */
    @Transactional
    public void checkOverdue() {
        List<BorrowRecord> records = borrowRecordRepository
            .findByStatusAndDueDateBefore(BorrowStatus.BORROWED, LocalDateTime.now());
        
        for (BorrowRecord record : records) {
            record.setStatus(BorrowStatus.OVERDUE);
            borrowRecordRepository.save(record);
            
            // 发送逾期通知
            User user = userService.getUserById(record.getUserId());
            notificationService.sendOverdueNotification(user, record);
            
            log.warn("借阅记录 {} 已逾期", record.getId());
        }
    }
    
    /**
     * 转换为响应DTO
     */
    private BorrowResponse convertToResponse(BorrowRecord record) {
        BorrowResponse response = new BorrowResponse();
        response.setId(record.getId());
        response.setBookId(record.getBookId());
        response.setBookTitle(record.getBookTitle());
        response.setBookIsbn(record.getBookIsbn());
        response.setBorrowDate(record.getBorrowDate());
        response.setDueDate(record.getDueDate());
        response.setReturnDate(record.getReturnDate());
        response.setStatus(record.getStatus().name());
        response.setRenewed(record.getRenewed());
        response.setRenewCount(record.getRenewCount());
        response.setOverdueDays(record.getOverdueDays());
        response.setFineAmount(record.getFineAmount());
        response.setFinePaid(record.getFinePaid());
        response.setNotes(record.getNotes());
        response.setCreatedAt(record.getCreatedAt());
        return response;
    }
}
```

```java
import com.library.model.User;
import com.library.repository.ReservationRecordRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.List;
import java.util.stream.Collectors;

@Slf4j
@Service
@RequiredArgsConstructor
public class ReservationService {
    
    private final ReservationRecordRepository reservationRecordRepository;
    private final BookService bookService;
    private final UserService userService;
    private final NotificationService notificationService;
    
    // 业务常量
    private static final int RESERVATION_EXPIRE_DAYS = 3;
    
    /**
     * 预约书籍
     */
    @Transactional
    public ReservationResponse reserveBook(Long userId, ReservationRequest request) {
        // 1. 验证用户
        User user = userService.getUserById(userId);
        if (user == null) {
            throw new IllegalArgumentException("用户不存在");
        }
        
        // 2. 验证书籍
        Book book = bookService.getBookById(request.getBookId());
        if (book == null) {
            throw new IllegalArgumentException("书籍不存在");
        }
        
        // 3. 检查是否已预约
        if (reservationRecordRepository.existsByUserIdAndBookIdAndStatus(
                userId, book.getId(), ReservationStatus.WAITING)) {
            throw new IllegalArgumentException("您已预约该书籍");
        }
        
        // 4. 计算队列位置
        List<ReservationRecord> queue = reservationRecordRepository
            .findByBookIdAndStatusOrderByCreatedAtAsc(book.getId(), ReservationStatus.WAITING);
        int queuePosition = queue.size() + 1;
        
        // 5. 创建预约记录
        ReservationRecord record = new ReservationRecord();
        record.setUserId(userId);
        record.setBookId(book.getId());
        record.setStudentId(user.getStudentId());
        record.setBookTitle(book.getTitle());
        record.setReservationDate(LocalDateTime.now());
        record.setExpireDate(LocalDateTime.now().plusDays(RESERVATION_EXPIRE_DAYS));
        record.setStatus(ReservationStatus.WAITING);
        record.setQueuePosition(queuePosition);
        
        ReservationRecord saved = reservationRecordRepository.save(record);
        
        // 6. 发送通知
        notificationService.sendReservationNotification(user, book, queuePosition);
        
        log.info("用户 {} 预约书籍 {}, 队列位置: {}", user.getStudentId(), book.getTitle(), queuePosition);
        
        return convertToResponse(saved);
    }
    
    /**
     * 取消预约
     */
    @Transactional
    public void cancelReservation(Long reservationId, String reason) {
        ReservationRecord record = reservationRecordRepository.findById(reservationId)
            .orElseThrow(() -> new IllegalArgumentException("预约记录不存在"));
        
        if (record.getStatus() != ReservationStatus.WAITING) {
            throw new IllegalArgumentException("只能取消等待中的预约");
        }
        
        record.setStatus(ReservationStatus.CANCELLED);
        record.setCancelReason(reason);
        reservationRecordRepository.save(record);
        
        // 更新后续队列位置
        updateQueuePositions(record.getBookId());
        
        log.info("预约 {} 已取消", reservationId);
    }
    
    /**
     * 书籍归还时通知预约用户
     */
    @Transactional
    public void notifyNextReservation(Long bookId) {
        List<ReservationRecord> queue = reservationRecordRepository
            .findByBookIdAndStatusOrderByCreatedAtAsc(bookId, ReservationStatus.WAITING);
        
        if (!queue.isEmpty()) {
            ReservationRecord next = queue.get(0);
            next.setStatus(ReservationStatus.AVAILABLE);
            next.setNotifyDate(LocalDateTime.now());
            reservationRecordRepository.save(next);
            
            // 发送到馆通知
            User user = userService.getUserById(next.getUserId());
            Book book = bookService.getBookById(bookId);
            notificationService.sendBookAvailableNotification(user, book);
            
            log.info("通知用户 {} 取书: {}", user.getStudentId(), book.getTitle());
        }
    }
    
    /**
     * 查询用户预约记录
     */
    public List<ReservationResponse> getUserReservations(Long userId) {
        List<ReservationRecord> records = reservationRecordRepository
            .findByUserIdOrderByCreatedAtDesc(userId);
        return records.stream()
            .map(this::convertToResponse)
            .collect(Collectors.toList());
    }
    
    /**
     * 检查过期预约（定时任务调用）
     */
    @Transactional
    public void checkExpiredReservations() {
        List<ReservationRecord> expired = reservationRecordRepository
            .findByStatusAndExpireDateBefore(ReservationStatus.AVAILABLE, LocalDateTime.now());
        
        for (ReservationRecord record : expired) {
            record.setStatus(ReservationStatus.EXPIRED);
            reservationRecordRepository.save(record);
            
            // 通知下一位
            notifyNextReservation(record.getBookId());
            
            log.info("预约 {} 已过期", record.getId());
        }
    }
    
    /**
     * 更新队列位置
     */
    private void updateQueuePositions(Long bookId) {
        List<ReservationRecord> queue = reservationRecordRepository
            .findByBookIdAndStatusOrderByCreatedAtAsc(bookId, ReservationStatus.WAITING);
        
        for (int i = 0; i < queue.size(); i++) {
            ReservationRecord record = queue.get(i);
            record.setQueuePosition(i + 1);
            reservationRecordRepository.save(record);
        }
    }
    
    /**
     * 转换为响应DTO
     */
    private ReservationResponse convertToResponse(ReservationRecord record) {
        ReservationResponse response = new ReservationResponse();
        response.setId(record.getId());
        response.setBookId(record.getBookId());
        response.setBookTitle(record.getBookTitle());
        response.setReservationDate(record.getReservationDate());
        response.setExpireDate(record.getExpireDate());
        response.setStatus(record.getStatus().name());
        response.setQueuePosition(record.getQueuePosition());
        response.setCreatedAt(record.getCreatedAt());
        return response;
    }
}
```

```vue
import { ref, onMounted } from 'vue'
import { borrowApi, type BorrowRecord } from '../api/borrowApi'

const activeTab = ref<'current' | 'history'>('current')
const currentBorrows = ref<BorrowRecord[]>([])
const borrowHistory = ref<BorrowRecord[]>([])

onMounted(() => {
  loadCurrentBorrows()
  loadBorrowHistory()

async function loadCurrentBorrows() {
  try {
    const response = await borrowApi.getCurrentBorrows()
    if (response.data.success) {
      currentBorrows.value = response.data.data
    }
  } catch (error) {

  }

  try {
    const response = await borrowApi.getBorrowHistory()
    if (response.data.success) {
      borrowHistory.value = response.data.data
    }
  } catch (error) {
    console.error('加载借阅历史失败:', error)
  }

async function handleRenew(recordId: number) {
  if (!confirm('确定要续借这本书吗?')) return
  
  try {
    const response = await borrowApi.renewBorrow(recordId)
    if (response.data.success) {
      alert('续借成功!')
      loadCurrentBorrows()
    } else {
      alert(response.data.message)
    }
  } catch (error: any) {
    alert(error.response?.data?.message || '续借失败')
  }
}

async function handleReturn(recordId: number) {
  if (!confirm('确定要归还这本书吗?')) return
  
  try {
    const response = await borrowApi.returnBook(recordId)
    if (response.data.success) {
      alert('归还成功!')
      loadCurrentBorrows()
      loadBorrowHistory()
    } else {
      alert(response.data.message)
    }
  } catch (error: any) {
    alert(error.response?.data?.message || '归还失败')
  }
}

function canRenew(borrow: BorrowRecord): boolean {
  return borrow.status === 'BORROWED' && 
         borrow.renewCount < 1 && 
         !isOverdue(borrow.dueDate)
}

function isOverdue(dueDate: string): boolean {
  return new Date(dueDate) < new Date()
}

function formatDate(dateStr: string): string {
  const date = new Date(dateStr)
  return date.toLocaleDateString('zh-CN')
}

function getStatusText(status: string): string {
  const statusMap: Record<string, string> = {
    PENDING: '待审核',
    APPROVED: '已批准',
    BORROWED: '借阅中',
    RETURNED: '已归还',
    OVERDUE: '已逾期',
    REJECTED: '已拒绝'
  }
  return statusMap[status] || status
}
```

#### MyReservations.vue (我的预约)

```vue
<template>
  <div class="my-reservations">
    <div class="page-header">
      <h1>我的预约</h1>
    </div>

    <div v-if="reservations.length === 0" class="empty-state">
      <p>暂无预约记录</p>
    </div>

    <div v-else class="reservation-list">
      <div 
        v-for="reservation in reservations" 
        :key="reservation.id"
        class="reservation-card">
        <div class="book-info">
          <h3>{{ reservation.bookTitle }}</h3>
          <div class="meta">
            <span class="queue">队列位置: 第{{ reservation.queuePosition }}位</span>
            <span :class="`status-${reservation.status.toLowerCase()}`">
              {{ getStatusText(reservation.status) }}
            </span>
          </div>
        </div>

        <div class="reservation-details">
          <div class="detail-item">
            <span class="label">预约日期:</span>
            <span>{{ formatDate(reservation.reservationDate) }}</span>
          </div>
          <div class="detail-item">
            <span class="label">过期日期:</span>
            <span>{{ formatDate(reservation.expireDate) }}</span>
          </div>
        </div>

        <div class="actions">
          <button 
            v-if="reservation.status === 'WAITING'"
            @click="handleCancel(reservation.id)"
            class="btn-cancel">
            取消预约
          </button>
          <button 
            v-if="reservation.status === 'AVAILABLE'"
            class="btn-pickup">
            去取书
          </button>
        </div>
      </div>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { reservationApi, type ReservationRecord } from '../api/borrowApi'

const reservations = ref<ReservationRecord[]>([])

onMounted(() => {
  loadReservations()
})

async function loadReservations() {
  try {
    const response = await reservationApi.getReservations()
    if (response.data.success) {
      reservations.value = response.data.data
    }
  } catch (error) {
    console.error('加载预约记录失败:', error)
  }
}

async function handleCancel(reservationId: number) {
  const reason = prompt('请输入取消原因（可选）:')
  if (reason === null) return // 用户点击了取消
  
  try {
    const response = await reservationApi.cancelReservation(reservationId, reason)
    if (response.data.success) {
      alert('预约已取消')
      loadReservations()
    } else {
      alert(response.data.message)
    }
  } catch (error: any) {
    alert(error.response?.data?.message || '取消失败')
  }
}

function formatDate(dateStr: string): string {
  const date = new Date(dateStr)
  return date.toLocaleDateString('zh-CN')
}

function getStatusText(status: string): string {
  const statusMap: Record<string, string> = {
    WAITING: '等待中',
    AVAILABLE: '可取书',
    COMPLETED: '已完成',
    CANCELLED: '已取消',
    EXPIRED: '已过期'
  }
  return statusMap[status] || status
}
</script>
```

## 📝 注意事项

### 业务规则
1. 用户最多同时借阅5本书
2. 默认借阅期限30天
3. 可续借1次，延长15天
4. 逾期罚款0.5元/天
5. 预约保留3天

### 技术要点
1. 使用JPA事务管理保证数据一致性
2. 定时任务使用@Scheduled注解
3. 状态机管理借阅和预约状态
4. 使用乐观锁防止并发问题

### 安全考虑
1. 所有API需要JWT认证
2. 用户只能操作自己的借阅和预约
3. 管理员才能审核借阅申请
